    labels = {
        "establishing context": "Provides background information or sets the scene.",
        "setup": "Introduces the situation or problem.",
        "escalation": "Increases tension or adds complexity.",
        "subversion": "Defies expectations or turns the situation on its head.",
        "callback": "Refers back to an earlier element in the joke.",
        "misdirection": "Leads the listener away from the punchline.",
        "timing": "Relies on the delivery or timing for its humor.",
        "meta-humor": "Reflects on the joke itself or the act of telling it.",
        "punchline": "The final, humorous revelation.",
        "redirection": "Changes the focus away from the punchline.",
        "non-line": "Does not fit into the other categories.",
        "wrap-up": "Provides closure or ties up loose ends.",
        "repetition": "Repeats information for emphasis or comedic effect."
    }

    labels_list = labels.values()
    classified_lines = []

    for line in joke_lines:
        for label in labels_list:
            if line.strip().lower().startswith(label):
                classified_lines.append(label)
                break

    if not classified_lines:
        classified_lines.append("non-line")

    return {"ANSWER": classified_lines}
