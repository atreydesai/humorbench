def classify_jokes(lines):
    labels = {
        "establishing_context": "Establishing context for the joke.",
        "setup": "Introducing the situation or characters for the punchline.",
        "escalation": "Building tension or complexity to heighten the punchline's impact.",
        "subversion": "Defying expectations or challenging the setup.",
        "callback": "Revisiting or echoing an earlier point in the joke.",
        "misdirection": "Distracting from the punchline's true meaning.",
        "timing": "Controlling when information is revealed to maximize the joke's effect.",
        "meta-humor": "Making fun of the joke itself, the joke-telling process, or the audience.",
        "punchline": "The final, often surprising or unexpected part of the joke.",
        "redirection": "Changing the focus of the joke to a different idea or object.",
        "non-line": "The joke doesn't follow a clear narrative or logical progression.",
        "wrap-up": "Summarizing or concluding the joke's elements for the audience.",
        "repetition": "Repeating information or ideas for emphasis or effect."
    }
    
    classified = []
    for line in lines:
        # Remove punctuation and convert to lowercase
        cleaned_line = re.sub(r'[^\w\s]', '', line).lower()
        
        # Find the most relevant label
        for label, description in labels.items():
            if description in cleaned_line:
                classified.append(label)
                break
    
    return {"ANSWER": classified}
